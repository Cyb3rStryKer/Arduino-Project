#include <Adafruit_NeoPixel.h>

//⚠️ATTENTION CODE UNIQUEMENT COMPATIBLE AVEC NEOPIXEL⚠️

#define NP_LED_COUNT_2 30
// ⚠️ Définir ici le nombre EXACT de LEDs de la Neopixel⚠️

#define PIN_SOUND_SENSOR_A0  A0
#define PIN_POTENTIOMETER_A1  A1
#define LED_PIN 2
//⚠️BRANCHER CAPTEUR DE SON SUR PIN A0, POTENTIOMETRE SUR A1 ET NEOPIXEL SUR D2⚠️

Adafruit_NeoPixel Neopixel_2(NP_LED_COUNT_2, LED_PIN, NEO_GRB + NEO_KHZ800);

uint16_t getAnalogMean(uint8_t pin, long n) {
  long sum = 0;
  for (int i = 0; i < n; i++) {
    sum += analogRead(pin);
    delay(1);
  }
  return sum / n;
}

int GetDecibels(int Analog_Sound)
{
  if (Analog_Sound <= 0) return 0;

  if (Analog_Sound <= 258)
    return round(1.8593 * exp(0.0135 * Analog_Sound));

  return round(16.777 * log(Analog_Sound) - 31.884);
}

void updateBar(int level)
{
  level = constrain(level, 0, NP_LED_COUNT_2);

  for (int i = 0; i < NP_LED_COUNT_2; i++)
  {
    if (i < level)
    {
      float ratio = (float)i / (NP_LED_COUNT_2 - 1);

      uint8_t r, g;

      if (ratio < 0.5f)
      {
        r = (uint8_t)(ratio * 2.0f * 255.0f);
        g = 255;
      }
      else
      {
        r = 255;
        g = (uint8_t)((1.0f - (ratio - 0.5f) * 2.0f) * 255.0f);
      }

      Neopixel_2.setPixelColor(i, Neopixel_2.Color(r, g, 0));
    }
    else
    {
      Neopixel_2.setPixelColor(i, Neopixel_2.Color(0, 0, 0));
    }
  }

  Neopixel_2.show();
}

void setup()
{
  Serial.begin(9600);

  Neopixel_2.begin();
  Neopixel_2.show();

  pinMode(PIN_SOUND_SENSOR_A0, INPUT);
  pinMode(PIN_POTENTIOMETER_A1, INPUT);

  updateBar(NP_LED_COUNT_2);
  delay(500);
  updateBar(0);
}

void loop()
{
  int analogSound = getAnalogMean(PIN_SOUND_SENSOR_A0, 5);
  int pot = analogRead(PIN_POTENTIOMETER_A1);

  float maxDb = 84.0 - (74.0 * pot / 1023.0);
  if (maxDb < 10.0) maxDb = 10.0;

  int db = GetDecibels(analogSound);
  if (db < 0) db = 0;

  int signal = constrain(
    (db * NP_LED_COUNT_2) / maxDb,
    0,
    NP_LED_COUNT_2
  );

  updateBar(signal);

  delay(30);
}
