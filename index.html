#include <Adafruit_NeoPixel.h>

// ========
// REGLAGES
// ========

// Nombre LEDs NEOPIXEL
#define NP_LED_COUNT_2 30

// Pin NEOPIXEL
#define LED_PIN 2

// Pin micro
#define PIN_SOUND_SENSOR_A0 A0

// Pin potentiometre
#define PIN_POTENTIOMETER_A1 A1

// Descente barre (ms)
#define FALL_TIME_MS 1000

// Attente avant descente (ms)
#define FALL_DELAY_MS 200

// Maintien pic (ms)
#define PEAK_HOLD_MS 3000

// Couleur pic
#define PEAK_R 255
#define PEAK_G 255
#define PEAK_B 255



// Compatible Vittascience
Adafruit_NeoPixel Neopixel_2(NP_LED_COUNT_2, LED_PIN, NEO_GRB + NEO_KHZ800);



int displayLevel = 0;

// Niveau du pic
int peakLevel = 0;


unsigned long lastRiseTime = 0;
unsigned long lastPeakTime = 0;
unsigned long lastFallUpdate = 0;



// ========
// MOYENNE MICRO
// ========

uint16_t getAnalogMean(byte pin, int n)
{
  long sum = 0;

  for(int i = 0; i < n; i++)
  {
    sum += analogRead(pin);
    delay(1);
  }

  return sum / n;
}



// ========
// CONVERSION SON EN DB
// ========

int GetDecibels(int value)
{

  if(value <= 0)
  {
    return 0;
  }


  if(value > 1023)
  {
    value = 1023;
  }


  int db;


  if(value <= 258)
  {
    db = (int)(1.8593 * exp(0.0135 * value));
  }
  else
  {
    db = (int)(16.777 * log(value) - 31.884);
  }


  if(db < 0)
  {
    db = 0;
  }


  if(db > 120)
  {
    db = 120;
  }


  return db;
}



// ========
// AFFICHAGE BARRE
// ========

void drawVU(int level, int peak)
{

  for(int i = 0; i < NP_LED_COUNT_2; i++)
  {

    // Barre LED
    if(i < level)
    {

      int r;
      int g;


      // Vert vers jaune
      if(i < 15)
      {
        r = map(i,0,14,0,255);
        g = 255;
      }

      // Jaune vers rouge
      else
      {
        r = 255;
        g = map(i,15,29,255,0);
      }


      Neopixel_2.setPixelColor(
        i,
        Neopixel_2.Color(r,g,0)
      );

    }

    else
    {
      Neopixel_2.setPixelColor(
        i,
        Neopixel_2.Color(0,0,0)
      );
    }

  }



  // Pic juste au-dessus de la barre
  if(peak >= 0 && peak < NP_LED_COUNT_2)
  {
    Neopixel_2.setPixelColor(
      peak,
      Neopixel_2.Color(PEAK_R,PEAK_G,PEAK_B)
    );
  }



  Neopixel_2.show();

}



// ========
// SETUP
// ========

void setup()
{

  Serial.begin(9600);


  pinMode(PIN_SOUND_SENSOR_A0,INPUT);
  pinMode(PIN_POTENTIOMETER_A1,INPUT);


  Neopixel_2.begin();

  Neopixel_2.show();


  drawVU(0,-1);

}



// ========
// LOOP
// ========

void loop()
{

  // Lecture micro
  int sound = getAnalogMean(PIN_SOUND_SENSOR_A0,5);



  // Lecture potentiometre
  int pot = analogRead(PIN_POTENTIOMETER_A1);



  // Sensibilité
  int maxDb = 84 - (14 * pot / 1023);


  if(maxDb < 10)
  {
    maxDb = 10;
  }



  int db = GetDecibels(sound);



  int signal = (db * NP_LED_COUNT_2) / maxDb;



  if(signal > NP_LED_COUNT_2)
  {
    signal = NP_LED_COUNT_2;
  }


  if(signal < 0)
  {
    signal = 0;
  }



  unsigned long now = millis();



  // Montée barre + nouveau pic
  if(signal > displayLevel)
  {
    displayLevel = signal;

    peakLevel = displayLevel;

    lastPeakTime = now;

    lastRiseTime = now;
  }



  // Descente progressive
  if(signal < displayLevel)
  {

    if(now - lastRiseTime > FALL_DELAY_MS)
    {

      if(now - lastFallUpdate > (FALL_TIME_MS / NP_LED_COUNT_2))
      {
  
      displayLevel--;

        lastFallUpdate = now;
      }

    }

  }



  // Après 3 secondes le pic revient
  if(now - lastPeakTime > PEAK_HOLD_MS)
  {
    peakLevel = displayLevel;
  }



  // Affichage
  drawVU(displayLevel,peakLevel);



  delay(10);

}
